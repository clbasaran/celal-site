import Firebase
import Combine
import SwiftUI

class RealtimeDataManager: ObservableObject {
    @Published var messages: [Message] = []
    @Published var isConnected: Bool = false
    
    private var cancellables = Set()
    private let db = Firestore.firestore()
    private var messageListener: ListenerRegistration?
    
    init() {
        setupRealtimeConnection()
    }
    
    func setupRealtimeConnection() {
        // Real-time message listener with advanced querying
        messageListener = db.collection("messages")
            .order(by: "timestamp", descending: false)
            .whereField("isActive", isEqualTo: true)
            .addSnapshotListener { [weak self] snapshot, error in
                guard let self = self else { return }
                
                if let error = error {
                    print("Firestore error: \(error.localizedDescription)")
                    self.isConnected = false
                    return
                }
                
                guard let documents = snapshot?.documents else {
                    self.isConnected = false
                    return
                }
                
                // Process document changes with sophisticated handling
                self.processDocumentChanges(snapshot?.documentChanges ?? [])
                self.isConnected = true
                
                // Reactive update with smooth animations
                withAnimation(.easeInOut(duration: 0.3)) {
                    self.messages = documents.compactMap { doc in
                        try? doc.data(as: Message.self)
                    }
                }
            }
    }
    
    private func processDocumentChanges(_ changes: [DocumentChange]) {
        for change in changes {
            switch change.type {
            case .added:
                // Handle new message with sound/haptic feedback
                HapticManager.shared.lightImpact()
                AudioManager.shared.playNotificationSound()
                
            case .modified:
                // Handle message updates
                print("Message updated: \(change.document.documentID)")
                
            case .removed:
                // Handle message deletion
                print("Message removed: \(change.document.documentID)")
            }
        }
    }
    
    func sendMessage(_ content: String) async throws {
        let message = Message(
            id: UUID().uuidString,
            content: content,
            senderId: UserManager.shared.currentUserId,
            timestamp: Date(),
            isActive: true
        )
        
        try await db.collection("messages")
            .document(message.id)
            .setData(from: message)
        
        // Optional: Trigger cloud function for push notifications
        await triggerNotificationFunction(for: message)
    }
    
    private func triggerNotificationFunction(for message: Message) async {
        // Call cloud function for push notifications
        let functions = Functions.functions()
        let data = [
            "messageId": message.id,
            "content": message.content,
            "senderId": message.senderId
        ]
        
        do {
            let result = try await functions.httpsCallable("sendMessageNotification").call(data)
            print("Notification sent: \(result.data)")
        } catch {
            print("Failed to send notification: \(error)")
        }
    }
    
    deinit {
        messageListener?.remove()
    }
}

Example: Smart Notification Function

// Cloud Function (Node.js)
const functions = require('firebase-functions');
const admin = require('firebase-admin');

exports.sendMessageNotification = functions.https.onCall(async (data, context) => {
    const { messageId, content, senderId } = data;
    
    // Get sender info
    const senderDoc = await admin.firestore()
        .collection('users')
        .doc(senderId)
        .get();
    
    const senderName = senderDoc.data()?.displayName || 'Someone';
    
    // Get all users except sender for notification
    const usersSnapshot = await admin.firestore()
        .collection('users')
        .where('id', '!=', senderId)
        .get();
    
    const tokens = [];
    usersSnapshot.forEach(doc => {
        const fcmToken = doc.data().fcmToken;
        if (fcmToken) tokens.push(fcmToken);
    });
    
    if (tokens.length === 0) return { success: false, message: 'No tokens found' };
    
    // Send multicast notification
    const message = {
        notification: {
            title: `New message from ${senderName}`,
            body: content.length > 50 ? content.substring(0, 50) + '...' : content,
        },
        data: {
            messageId: messageId,
            senderId: senderId,
            type: 'new_message'
        },
        tokens: tokens
    };
    
    try {
        const response = await admin.messaging().sendMulticast(message);
        return { 
            success: true, 
            successCount: response.successCount,
            failureCount: response.failureCount 
        };
    } catch (error) {
        throw new functions.https.HttpsError('internal', 'Failed to send notification');
    }
});

Smart Analytics Implementation

import FirebaseAnalytics

class AnalyticsManager {
    static let shared = AnalyticsManager()
    
    private init() {}
    
    // Track user journey with sophisticated events
    func trackMessageSent(content: String, recipientCount: Int) {
        Analytics.logEvent("message_sent", parameters: [
            "content_length": content.count,
            "recipient_count": recipientCount,
            "message_type": determineMessageType(content),
            "timestamp": Date().timeIntervalSince1970
        ])
    }
    
    func trackUserEngagement(feature: String, duration: TimeInterval) {
        Analytics.logEvent("feature_engagement", parameters: [
            "feature_name": feature,
            "engagement_duration": duration,
            "session_id": SessionManager.shared.currentSessionId
        ])
    }
    
    // Advanced user properties for segmentation
    func setUserProperties(for user: User) {
        Analytics.setUserProperty(user.subscriptionTier, forName: "subscription_tier")
        Analytics.setUserProperty(user.preferredLanguage, forName: "preferred_language")
        Analytics.setUserProperty(String(user.daysActive), forName: "days_active")
        
        // Custom computed properties
        Analytics.setUserProperty(
            user.messagesSent > 100 ? "power_user" : "regular_user", 
            forName: "user_category"
        )
    }
    
    private func determineMessageType(_ content: String) -> String {
        if content.contains("http") || content.contains("www") {
            return "link_sharing"
        } else if content.count > 200 {
            return "long_message"
        } else if content.contains("@") {
            return "mention"
        }
        return "regular"
    }
}
